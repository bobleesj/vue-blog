<!--  Template  -->
<template>
  <div class="upload">
    <h1>Upload</h1>
    <textarea v-model="markdown" class="markdown" placeholder="Preview"></textarea>
    <div v-html="renderedMarkdown"></div>
    <button v-on:click="counter += 1">Increase Counter</button>
    <p>{{counter}}</p>
    <button v-on:click="addNewTodo()">Upload Blog</button>
    <p>{{posts}}</p>
  </div>
</template>
 
<!--  Vue  -->
<script>
import showdown from 'Showdown'
var converter = new showdown.Converter()
 
export default {
  name: 'upload',
  data: function () {
    return {
      markdown: '',
      counter: 0,
      posts: []
    }
  },
  computed: {
    renderedMarkdown: function () {
      var text = this.markdown
      var html = converter.makeHtml(text)
      return html
    }
  },
  methods: {
    addNewTodo: function () {
      this.posts.push({
        title: 'Title',
        subtitle: 'Subtitle'
      })
    }
  }
}
</script>
